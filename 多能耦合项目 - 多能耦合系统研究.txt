$$C_i = \frac{\int_0^T |\Delta P_i(t)| dt}{\int_0^T |\Delta P_{dist}(t)| dt}$$

动态耦合增益：输入（电侧扰动）到输出（氢压/蒸汽压力）在给定频段的增益
传播时延与路径贡献：从扰动发生到各子系统响应峰值的时间差，配合贡献分解识别主通道
1.4 指标体系：你需要一套“第1条专用”的动态量化指标

建议按“响应质量 + 耦合强度 + 非线性阈值”三类指标组织：

A. 动态响应质量（每个能流都可算）
峰值偏差（超调/欠调）：$\Delta y_{max}$
恢复时间：$T_{rec}$（回到±ε带宽）
累积偏差：$\int_0^T |y(t)-y^\*| dt$
超限风险：是否触发约束/保护（次数、持续时间）
B. 跨域耦合强度（重点体现“交互”）
电→氢耦合系数：$K_{e\to h} = \frac{\Delta P_{e\to h}}{\Delta P_{dist}}$（或频域增益）
电→热耦合系数：$K_{e\to t}$
反向耦合：$K_{h\to e}$、$K_{t\to e}$（比如压缩机功率上升导致电侧再扰动）
C. 非线性阈值与模式切换指标（体现“非线性”）
约束触发阈值：扰动幅值/持续时间达到多少会触发电解槽限功率、储罐Pmin、蒸汽母管下限等
模式切换图谱：不同工作区间下响应从“线性平滑”变成“突变/振荡”的边界
饱和敏感度：控制器限幅前后指标的突变比（例如恢复时间放大倍数）

这些指标后面可直接服务第3条“灵活性评估”，形成上下衔接。

function run = simulate_multiscale(p, cfg, sc)
% 说明：
% - 快层：电侧功率/频率等效动态（秒级甚至更快）
% - 慢层：热惯性、储氢压力/库存（分钟级）
% - 实现方式：多速率积分（Fast dt_f, Slow dt_s），接口变量按 step-and-hold 或线性外推交换

t0 = 0;
tf = cfg.sim.Tend;
dt_f = cfg.sim.dt_fast;
dt_s = cfg.sim.dt_slow;

% 初值（工作点）
x = cfg.init.x0(:);

% 日志
tlog = t0:dt_f:tf;
Xlog = zeros(numel(x), numel(tlog));
Ylog = zeros(numel(model_outputs(x, p)), numel(tlog));
Ulog = zeros(5, numel(tlog)); % [Pws, dLe, dLsteam, dH2, modeFlags...] 简化

idx = 1;

% 慢变量缓存（用于快层参数化）
slowState = extract_slow_state(x, cfg);

t = t0;
nextSlowUpdate = t0;

while t <= tf + 1e-12
    % 生成扰动输入（按当前时间）
    u = sc.u(t);

    % 慢层更新：每 dt_s 更新一次慢变量/接口
    if t >= nextSlowUpdate - 1e-12
        slowState = extract_slow_state(x, cfg);
        nextSlowUpdate = nextSlowUpdate + dt_s;
    end

    % 快层一步（显式RK4）
    x = rk4_step(@(tt,xx) model_dynamics(tt,xx,u,slowState,p,cfg), t, x, dt_f);

    % 记录
    Xlog(:,idx) = x;
    y = model_outputs(x, p);
    Ylog(:,idx) = y;
    Ulog(:,idx) = pack_u(u);

    t = t + dt_f;
    idx = idx + 1;
end

% trim
tlog = tlog(1:idx-1);
Xlog = Xlog(:,1:idx-1);
Ylog = Ylog(:,1:idx-1);
Ulog = Ulog(:,1:idx-1);

run = struct();
run.name = sc.name;
run.t = tlog;
run.X = Xlog;
run.Y = Ylog;
run.U = Ulog;
run.scenario = sc;
run.cfg = cfg;
run.p = p;
end

function xnext = rk4_step(f, t, x, h)
k1 = f(t, x);
k2 = f(t + 0.5*h, x + 0.5*h*k1);
k3 = f(t + 0.5*h, x + 0.5*h*k2);
k4 = f(t + h, x + h*k3);
xnext = x + (h/6)*(k1 + 2*k2 + 2*k3 + k4);
end

function slow = extract_slow_state(x, cfg)
% 约定：慢变量索引在 cfg.idxSlow
slow = x(cfg.idxSlow);
end

function dx = model_dynamics(t, x, u, slowState, p, cfg)
% 状态 x = [delta_f; P_bat; SOC; P_elz; mH2; P_heat; E_th; xi]
% 输入 u: Pws, dLe, dLsteam, dH2dem
%
% 非线性点：
% - 约束触发导致分配比例变化/功率限幅
% - 电解槽效率随功率非线性变化
% - 热储上下限、SOC上下限
% - 氢库存上下限、库存影响“可调度性”

delta_f = x(1);
Pbat    = x(2);
SOC     = x(3);
Pelz    = x(4);
mH2     = x(5);
Pheat   = x(6);
Eth     = x(7);
xi      = x(8);

% 扰动
Pws     = u.Pws;       % 可再生净出力扰动（+供给增加）
dLe     = u.dLe;       % 电负荷扰动（+负荷增加）
dLsteam = u.dLsteam;   % 蒸汽/热负荷扰动（+热需求增加）
dH2dem  = u.dH2dem;    % 氢需求扰动（+需求增加）

% -------- 1) 功率缺口（电侧平衡）--------
% 简化：Pgrid(等效机组) + Pws + Pbat - Pelz - Pheat - Le = 0
% 用频率动态反映电侧功率不平衡
% Pimb = Pws - dLe + Pbat - Pelz - Pelec_heat - Pcomp(压缩耗能的电功率)
% 其中 Pheat 是热功率输出，电侧消耗 Pelec_heat = Pheat/COP

% 电解槽产氢率（非线性效率）
eff_elz = p.h2.eff_a - p.h2.eff_b*((Pelz/p.lim.Pelz_max)-0.5)^2;
eff_elz = max(0.45, min(0.80, eff_elz));
mH2_prod = p.h2.kH2 * eff_elz * Pelz;  % kg/s

% 氢系统“流量”简化：需求 = base + dH2dem（这里 base=0）
mH2_dem = max(0, dH2dem);              % kg/s（作为扰动输入）
Pcomp   = p.h2.kComp * mH2_dem;        % MW 压缩耗能（简化）

% 电->热电功率消耗
Pelec_heat = Pheat / max(0.1, p.heat.COP);

% 电侧功率不平衡
Pimb = (Pws - dLe) + Pbat - Pelz - Pelec_heat - Pcomp;

% 频率动态（等效摆动）
ddelta_f = (1/p.elec.M) * (Pimb - p.elec.D*delta_f);

% -------- 2) 控制：频差调节 + 积分，分配到三类可控主体 --------
% 控制目标：抑制 delta_f => 生成调节功率 Pref
% Pref 正表示需要“增加供给/减少负荷”（即填补缺口）
Pref = -p.ctrl.Kf*delta_f - p.ctrl.Ki*xi;

% 缺口分配（考虑约束动态调整）
[ab, ae, ah, flags] = allocation_with_constraints(SOC, mH2, Eth, p, cfg);

Pbat_ref  = ab * Pref;
Pelz_ref  = -ae * Pref;   % 注意：电解槽是“可调负荷”，Pref>0时应降负荷 => Pelz_ref 负号
Pheat_ref = -ah * Pref;   % 同理，热侧电耗

% 限幅
Pbat_ref  = sat(Pbat_ref,  -p.lim.Pbat_max, p.lim.Pbat_max);
Pelz_ref  = sat(Pelz + Pelz_ref, p.lim.Pelz_min, p.lim.Pelz_max); % 以当前为基准调
Pheat_ref = sat(Pheat + Pheat_ref, p.lim.Pheat_min, p.lim.Pheat_max);

% -------- 3) 动态执行：一阶跟随 --------
dPbat  = (1/p.bat.Tp)   * (Pbat_ref - Pbat);
dPelz  = (1/p.h2.Telz)  * (Pelz_ref - Pelz);
dPheat = (1/p.heat.Theat)* (Pheat_ref - Pheat);

% -------- 4) SOC 动态（MWh -> MW 转换）--------
% SOCdot = -Pbat / Ecap (放电SOC下降), 充电SOC上升
% 这里以 MW，Ecap(MWh) => SOCdot = -Pbat/(Ecap*3600)
if Pbat >= 0
    Pbat_eff = Pbat / p.bat.eta_dis;
else
    Pbat_eff = Pbat * p.bat.eta_ch;
end
dSOC = -(Pbat_eff)/(p.bat.Ecap*3600);

% SOC限幅处理（碰到边界则禁止继续越界）
[dSOC, Pbat, dPbat] = enforce_soc_limits(SOC, dSOC, Pbat, dPbat, p);

% -------- 5) 氢库存动态（kg）--------
dmH2 = mH2_prod - mH2_dem;
[dmH2, dPelz] = enforce_h2_limits(mH2, dmH2, Pelz, dPelz, p);

% -------- 6) 热储动态（MWhth）--------
% 热储能量变化：Ethdot = (Pheat - Lsteam)*dt - loss
dEth = (Pheat - dLsteam)/(3600) - p.heat.kLoss*Eth;
[dEth, dPheat] = enforce_eth_limits(Eth, dEth, Pheat, dPheat, p);

% -------- 7) 控制积分 --------
dxi = delta_f;

dx = [ddelta_f; dPbat; dSOC; dPelz; dmH2; dPheat; dEth; dxi];

% 输出 flags 到 u.flags（用于记录可选）

    ab = 0.15*ab; flags = flags + 1;
elseif SOC >= cfg.limits.SOC_max - 0.02
    ab = 0.15*ab; flags = flags + 2;
end

end

if Eth <= cfg.limits.Eth_min + 20
    ah = 0.10*ah; flags = flags + 8;
end

% 归一化
s = ab + ae + ah;
ab = ab/s; ae = ae/s; ah = ah/s;
end

end
end

function E = energy_decomposition(run, p, cfg)
% 功率/能量分解：回答“谁吸收扰动、吸收了多少”
t = run.t;
Y = run.Y;
U = run.U;

delta_f = Y(1,:);
Pbat    = Y(2,:);
Pelz    = Y(3,:);
Pheat   = Y(4,:);

Pws  = U(1,:);
dLe  = U(2,:);
dLsteam = U(3,:);
dH2dem  = U(4,:);

% 压缩耗能功率（与需求相关）
Pcomp = p.h2.kComp * max(0, dH2dem);

% 电侧热耗电
Pelec_heat = Pheat / max(0.1, p.heat.COP);

% 扰动功率：净扰动 = Pws - dLe （正为有利）
Pdist = Pws - dLe;

% 系统功率分配项（按Pimb方程拆解）
P_terms = struct();
P_terms.grid_like = -(p.elec.D*delta_f); % 等效“网侧/机组”响应项（从频差映射）
P_terms.battery   = Pbat;
P_terms.elz_load  = -Pelz;
P_terms.heat_load = -Pelec_heat;
P_terms.comp_load = -Pcomp;

% 贡献度量：用 |Pi| 对 |Pdist| 的积分比值
dt = mean(diff(t));
den = trapz(t, abs(Pdist)) + 1e-9;

fields = fieldnames(P_terms);
C = struct();
for i=1:numel(fields)
    f = fields{i};
    C.(f) = trapz(t, abs(P_terms.(f))) / den;
end

% 能量缓冲贡献：电池能量变化/热储变化（以绝对变化积分）
SOC = Y(5,:);
mH2 = Y(6,:);
Eth = Y(7,:);

% 绝对变化
C.dSOC = trapz(t, abs(gradient(SOC,dt))) / (trapz(t, abs(Pdist))/ (p.bat.Ecap*3600) + 1e-9);
C.dmH2 = trapz(t, abs(gradient(mH2,dt))) / (trapz(t, abs(Pdist)) + 1e-9);
C.dEth = trapz(t, abs(gradient(Eth,dt))) / (trapz(t, abs(Pdist)) + 1e-9);

E = struct();
E.Pdist = Pdist;
E.Pterms = P_terms;
E.contrib = C;
end

function C = coupling_identify(run, p, cfg)
% 耦合识别：回答“电扰动如何传到氢/热”“耦合增益/时延/频带”
t = run.t;
Y = run.Y;
U = run.U;

Pdist = U(1,:) - U(2,:); % Pws - dLe
mH2   = Y(6,:);
Eth   = Y(7,:);
delta_f = Y(1,:);

dt = mean(diff(t));
fs = 1/dt;

% 去均值
xd = detrend(Pdist(:));
yh = detrend(mH2(:));
yt = detrend(Eth(:));
yf = detrend(delta_f(:));

% 互相关找传播时延（粗略）
[maxc_h, lag_h] = max(xcorr(yh, xd, 'coeff'));
[maxc_t, lag_t] = max(xcorr(yt, xd, 'coeff'));
[maxc_f, lag_f] = max(xcorr(yf, xd, 'coeff'));

lagSec_h = (lag_h - numel(xd)) / fs;
lagSec_t = (lag_t - numel(xd)) / fs;
lagSec_f = (lag_f - numel(xd)) / fs;

% 频域耦合：计算在指定频段的功率谱比（粗略增益）
bands = cfg.coupling.fbands;
G = zeros(size(bands,1), 3); % [toH2,toHeat,toFreq]
for i=1:size(bands,1)
    f1 = bands(i,1); f2 = bands(i,2);
    [Pxx, f] = periodogram(xd, [], [], fs);
    [Phh, ~] = periodogram(yh, [], [], fs);
    [Ptt, ~] = periodogram(yt, [], [], fs);
    [Pff, ~] = periodogram(yf, [], [], fs);

    idx = (f>=f1) & (f<=f2);
    G(i,1) = sqrt(trapz(f(idx), Phh(idx)) / (trapz(f(idx), Pxx(idx))+1e-12));
    G(i,2) = sqrt(trapz(f(idx), Ptt(idx)) / (trapz(f(idx), Pxx(idx))+1e-12));
    G(i,3) = sqrt(trapz(f(idx), Pff(idx)) / (trapz(f(idx), Pxx(idx))+1e-12));
end

C = struct();
C.lagSec = struct('toH2',lagSec_h,'toHeat',lagSec_t,'toFreq',lagSec_f);
C.maxCorr = struct('toH2',maxc_h,'toHeat',maxc_t,'toFreq',maxc_f);
C.bandGain = G;
C.bands = bands;
end

function M = metrics_compute(run, p, cfg)
% 指标：响应质量 + 超限/触发 + 恢复时间
t = run.t;
Y = run.Y;

delta_f = Y(1,:);
SOC     = Y(5,:);
mH2     = Y(6,:);
Eth     = Y(7,:);

% 频率指标
M.freq.maxAbs = max(abs(delta_f));
M.freq.rms    = rms(delta_f);

% 恢复时间：回到±eps*max(|delta_f|) 或固定阈值
eps = cfg.metrics.band_eps;
thr = max(0.02, eps*max(abs(delta_f))+1e-9);
M.freq.recoveryTime = recovery_time(t, abs(delta_f), thr);

% 约束触发（超限次数/时长）
[M.soc] = limit_stats(t, SOC, p.lim.SOC_min, p.lim.SOC_max);
[M.h2 ] = limit_stats(t, mH2, p.lim.mH2_min, p.lim.mH2_max);
[M.eth] = limit_stats(t, Eth, p.lim.Eth_min, p.lim.Eth_max);

% 非线性“突变”探测：恢复时间、峰值随工况的变化可用于阈值地图（这里只输出基础量）
M.nonlinear.signature = [M.freq.maxAbs, M.freq.recoveryTime, M.soc.timeOut, M.h2.timeOut, M.eth.timeOut];
end

function Tr = recovery_time(t, xabs, thr)
% 最早时刻之后一直保持在阈值内
Tr = NaN;
N = numel(t);
for k=1:N
    if all(xabs(k:end) <= thr)
        Tr = t(k);
        return;
    end
end
end

function S = limit_stats(t, x, xmin, xmax)
out = (x<xmin) | (x>xmax);
dt = mean(diff(t));
S.timeOut = sum(out)*dt;
S.countCross = count_crossings(out);
S.minVal = min(x);
S.maxVal = max(x);
end

function n = count_crossings(b)
n = 0;
for k=2:numel(b)
    if b(k) && ~b(k-1)
        n = n + 1;
    end
end
end

function run = simulate_for_map(p, cfg, sc)
% 简化版仿真调用（复用 main 中的流程，但不重复写文件依赖）
t0 = 0; tf = cfg.sim.Tend;
dt_f = cfg.sim.dt_fast; dt_s = cfg.sim.dt_slow;

x = cfg.init.x0(:);
tlog = t0:dt_f:tf;
Xlog = zeros(numel(x), numel(tlog));
Ylog = zeros(numel(model_outputs(x,p)), numel(tlog));
Ulog = zeros(5, numel(tlog));

idx = 1;
slowState = x(cfg.idxSlow);
t = t0; nextSlowUpdate = t0;

while t <= tf + 1e-12
    u = sc.u(t);
    if t >= nextSlowUpdate - 1e-12
        slowState = x(cfg.idxSlow);
        nextSlowUpdate = nextSlowUpdate + dt_s;
    end
    x = rk4_step(@(tt,xx) model_dynamics(tt,xx,u,slowState,p,cfg), t, x, dt_f);

    Xlog(:,idx)=x;
    Ylog(:,idx)=model_outputs(x,p);
    Ulog(:,idx)=[u.Pws;u.dLe;u.dLsteam;u.dH2dem;u.flags];

    t=t+dt_f; idx=idx+1;
end

run = struct('t',tlog(1:idx-1),'X',Xlog(:,1:idx-1),'Y',Ylog(:,1:idx-1),'U',Ulog(:,1:idx-1));
end

function xnext = rk4_step(f, t, x, h)
k1 = f(t, x);
k2 = f(t + 0.5*h, x + 0.5*h*k1);
k3 = f(t + 0.5*h, x + 0.5*h*k2);
k4 = f(t + h, x + h*k3);
xnext = x + (h/6)*(k1 + 2*k2 + 2*k3 + k4);
end

function Kmodel = koopman_edmd(run, cfg)
% EDMD: 以 x_k -> x_{k+1} 拟合 Koopman 近似
% 可用于模态分析、预测、非线性映射的“线性解释”
%
% 这里用输出Y做状态，字典用 [y; y.^2; sin(y); 1]
Y = run.Y';
dt = mean(diff(run.t));

% 下采样以减小维度（可调）
ds = 10;
Y = Y(1:ds:end,:);
Y1 = Y(1:end-1,:);
Y2 = Y(2:end,:);

Phi1 = dict(Y1);
Phi2 = dict(Y2);

% 最小二乘解：Phi2 ≈ Phi1 * A'
A = (Phi1 \ Phi2); % A: maps Phi1 -> Phi2

% Koopman 模态（对A做特征分解）
[W, D] = eig(A);
lambda = diag(D);
omega = log(lambda)/dt;

Kmodel = struct();
Kmodel.A = A;
Kmodel.lambda = lambda;
Kmodel.omega = omega;
Kmodel.W = W;
Kmodel.dict = 'poly2+sin+const';
end

function Phi = dict(Y)
% Y: N x ny
Phi = [Y, Y.^2, sin(Y), ones(size(Y,1),1)];
end

function run = simulate_multiscale_internal(p, cfg, sc)

t0 = 0;
tf = cfg.sim.Tend;
dt_f = cfg.sim.dt_fast;
dt_s = cfg.sim.dt_slow;

x = cfg.init.x0(:);

tlog = t0:dt_f:tf;
Xlog = zeros(numel(x), numel(tlog));
Ylog = zeros(numel(model_outputs(x, p)), numel(tlog));
Ulog = zeros(5, numel(tlog));

idx = 1;
slowState = x(cfg.idxSlow);
t = t0;
nextSlowUpdate = t0;

while t <= tf + 1e-12
    u = sc.u(t);

    if t >= nextSlowUpdate - 1e-12
        slowState = x(cfg.idxSlow);
        nextSlowUpdate = nextSlowUpdate + dt_s;
    end

    x = rk4_step(@(tt,xx) model_dynamics(tt,xx,u,slowState,p,cfg), t, x, dt_f);

    Xlog(:,idx)=x;
    Ylog(:,idx)=model_outputs(x,p);
    Ulog(:,idx)=[u.Pws;u.dLe;u.dLsteam;u.dH2dem;u.flags];

    t=t+dt_f; idx=idx+1;
end

run = struct();
run.name = sc.name;
run.t = tlog(1:idx-1);
run.X = Xlog(:,1:idx-1);
run.Y = Ylog(:,1:idx-1);
run.U = Ulog(:,1:idx-1);
end

function xnext = rk4_step(f, t, x, h)
k1 = f(t, x);
k2 = f(t + 0.5*h, x + 0.5*h*k1);
k3 = f(t + 0.5*h, x + 0.5*h*k2);
k4 = f(t + h, x + h*k3);
xnext = x + (h/6)*(k1 + 2*k2 + 2*k3 + k4);
end

function PB = interface_power_balance(t, x, u, p, cfg)
% INTERFACE_POWER_BALANCE
% 统一输出电侧功率平衡各项，用于：
% - 第1条：动态响应机理量化（贡献分解、耦合通道）
% - 与第2条模型衔接：只要保证这里口径一致，后续模型可替换
%
% 返回 PB 结构体字段（示例）：
% PB.Pdist     = Pws - dLe
% PB.Pbat      = Pbat
% PB.Pelz      = -Pelz   (负号表示电侧消耗)
% PB.PelecHeat = -Pheat/COP
% PB.Pcomp     = -kComp*dH2dem
% PB.Pimb      = Pdist + Pbat + ... (合成不平衡)
% PB.grid_like = -D*df   (等效网侧/机组响应项)

y = model_outputs(x, p);
df    = y(1);
Pbat  = y(2);
Pelz  = y(3);
Pheat = y(4);

Pws = u.Pws;
dLe = u.dLe;
dH2 = max(0, u.dH2dem);

Pdist = Pws - dLe;

Pcomp = p.h2.kComp * dH2;                 % MW
PelecHeat = Pheat / max(0.1, p.heat.COP); % MW

PB = struct();
PB.t = t;
PB.Pdist = Pdist;

PB.Pbat = Pbat;
PB.Pelz = -Pelz;
PB.PelecHeat = -PelecHeat;
PB.Pcomp = -Pcomp;

PB.grid_like = -(p.elec.D * df);

PB.Pimb = PB.Pdist + PB.Pbat + PB.Pelz + PB.PelecHeat + PB.Pcomp + PB.grid_like;
end

function out = dist_utils(cmd, varargin)
% DIST_UTILS 扰动工具箱（第1条扰动库构建常用组件）
% 用法示例：
%   x = dist_utils('colored_noise', N, dt, 'low'); % 低频主导噪声
%   y = dist_utils('ramp', t, t0, dur, A);         % 斜坡
%   y = dist_utils('step', t, t0, dur, A);         % 方波阶跃（持续dur）
%   y = dist_utils('clip', x, xmin, xmax);

switch lower(cmd)
    case 'step'
        t  = varargin{1}; t0 = varargin{2}; dur = varargin{3}; A = varargin{4};
        out = A * ((t>=t0) && (t<=t0+dur));

    case 'ramp'
        t  = varargin{1}; t0 = varargin{2}; dur = varargin{3}; A = varargin{4};
        if t < t0
            out = 0;
        elseif t > t0 + dur
            out = A;
        else
            out = A * (t - t0)/dur;
        end

    case 'clip'
        x = varargin{1}; xmin = varargin{2}; xmax = varargin{3};
        out = min(max(x, xmin), xmax);

    case 'colored_noise'
        % 生成有色噪声：用一阶低通/高通方式塑形
        N  = varargin{1};
        dt = varargin{2};
        mode = varargin{3}; % 'low' or 'high'
        w = randn(N,1);

        switch lower(mode)
            case 'low'
                tau = 60; % 60s低通（可调）
                a = exp(-dt/tau);
                x = zeros(N,1);
                for k=2:N
                    x(k) = a*x(k-1) + (1-a)*w(k);
                end
                out = x;

            case 'high'
                tau = 20; % 20s高通（可调）
                a = exp(-dt/tau);
                lp = zeros(N,1);
                for k=2:N
                    lp(k) = a*lp(k-1) + (1-a)*w(k);
                end
                out = w - lp;

            otherwise
                out = w;
        end

    otherwise
        error('dist_utils: unknown cmd %s', cmd);
end
end

function LM = lin_local_modal(p, cfg, x_op, u_op)
% LIN_LOCAL_MODAL
% 在工作点 (x_op,u_op) 附近对 model_dynamics 进行数值线性化：
%   dx = f(x,u)  =>  d(δx) = A δx + B δu
% 输出模态：eig(A)，阻尼/频率，及对关键状态的参与因子（简化）
%
% 用法：
%   u0 = struct('Pws',0,'dLe',0,'dLsteam',0,'dH2dem',0,'flags',0);
%   LM = lin_local_modal(p,cfg,cfg.init.x0,u0);

if nargin < 4 || isempty(u_op)
    u_op = struct('Pws',0,'dLe',0,'dLsteam',0,'dH2dem',0,'flags',0);
end

x_op = x_op(:);

% 慢变量作为参数（快慢分解时的“冻结慢变量”）
slowState = x_op(cfg.idxSlow);

% 数值线性化
nx = numel(x_op);
nu = 4; % [Pws,dLe,dLsteam,dH2dem]（flags不作为输入）

epsx = 1e-6;
epsu = 1e-6;

f0 = model_dynamics(0, x_op, u_op, slowState, p, cfg);

A = zeros(nx,nx);
for i=1:nx
    dx = zeros(nx,1); dx(i) = epsx;
    fp = model_dynamics(0, x_op+dx, u_op, slowState, p, cfg);
    fm = model_dynamics(0, x_op-dx, u_op, slowState, p, cfg);
    A(:,i) = (fp - fm)/(2*epsx);
end

B = zeros(nx,nu);
for j=1:nu
    up = u_op; um = u_op;
    switch j
        case 1, up.Pws = up.Pws + epsu; um.Pws = um.Pws - epsu;
        case 2, up.dLe = up.dLe + epsu; um.dLe = um.dLe - epsu;
        case 3, up.dLsteam = up.dLsteam + epsu; um.dLsteam = um.dLsteam - epsu;
        case 4, up.dH2dem = up.dH2dem + epsu; um.dH2dem = um.dH2dem - epsu;
    end
    fp = model_dynamics(0, x_op, up, slowState, p, cfg);
    fm = model_dynamics(0, x_op, um, slowState, p, cfg);
    B(:,j) = (fp - fm)/(2*epsu);
end

% 模态
[V,D] = eig(A);
eigA = diag(D);

% 连续系统：lambda = sigma + j*omega
sigma = real(eigA);
omega = imag(eigA);
freqHz = abs(omega)/(2*pi);

% 阻尼比（对共轭对）
zeta = zeros(size(eigA));
for k=1:numel(eigA)
    if abs(omega(k)) > 1e-9
        zeta(k) = -sigma(k)/sqrt(sigma(k)^2 + omega(k)^2);
    else
        zeta(k) = NaN;
    end
end

% 参与因子（简化：|V(i,k)| 作为状态 i 对模态 k 的幅值贡献）
PF = abs(V);

LM = struct();
LM.A = A; LM.B = B;
LM.eigs = eigA;
LM.sigma = sigma;
LM.omega = omega;
LM.freqHz = freqHz;
LM.zeta = zeta;
LM.participation = PF;

% 建议：输出最关键模态
[~,idx] = sort(freqHz,'descend');
LM.keyModes = table(eigA(idx), sigma(idx), freqHz(idx), zeta(idx), ...
    'VariableNames', {'eig','sigma','freqHz','zeta'});

end

function s = safe_struct_merge(base, override)
% SAFE_STRUCT_MERGE 递归合并结构体 override -> base
% - base 作为默认配置
% - override 覆盖同名字段
% - 若字段都是struct，则递归合并
%
% 示例：
% cfg = safe_struct_merge(cfg, struct('sim',struct('Tend',3600)));

if isempty(base), s = override; return; end
if isempty(override), s = base; return; end

if ~isstruct(base) || ~isstruct(override)
    s = override;
    return;
end

s = base;
fn = fieldnames(override);

for i=1:numel(fn)
    f = fn{i};
    if isfield(base, f)
        if isstruct(base.(f)) && isstruct(override.(f))
            s.(f) = safe_struct_merge(base.(f), override.(f));
        else
            s.(f) = override.(f);
        end
    else
        s.(f) = override.(f);
    end
end
end

function BP = butter_bandpower(x, fs, f1, f2, order)
% BUTTER_BANDPOWER
% 对信号 x 进行 Butterworth 带通滤波并计算该频带功率（均方值）
% 用于第1条：不同时间尺度（分钟级/秒级）的耦合强度、扰动谱能量
%
% BP.band_rms : 频带RMS
% BP.band_power : 频带功率(=mean(x_band.^2))
% BP.x_band : 滤波后信号
%
% 注意：若频带超出 Nyquist 或过窄，会自动保护。

if nargin < 5 || isempty(order), order = 4; end

x = x(:);
if numel(x) < 10
    BP = struct('band_rms',NaN,'band_power',NaN,'x_band',x);
    return;
end

nyq = fs/2;
f1 = max(1e-6, min(f1, nyq*0.999));
f2 = max(1e-6, min(f2, nyq*0.999));
if f2 <= f1
    BP = struct('band_rms',NaN,'band_power',NaN,'x_band',zeros(size(x)));
    return;
end

Wn = [f1 f2]/nyq;
[b,a] = butter(order, Wn, 'bandpass');

% filtfilt 避免相位延迟
x_band = filtfilt(b,a, detrend(x));

band_power = mean(x_band.^2);
band_rms = sqrt(band_power);

BP = struct();
BP.band_rms = band_rms;
BP.band_power = band_power;
BP.x_band = x_band;
end

function C = coupling_identify(run, p, cfg)

% 1) 自动选择合适的输入通道（Pdist优先，否则用dLe/dLsteam/dH2dem）
% 2) 正确计算时延（秒），正值表示“输出滞后输入”
% 3) 输入无变化时避免NaN

Y = run.Y;
U = run.U;

% 候选输入（按优先级）

function [xin, name] = pick_in_signal(Pdist, dLe, dLsteam, dH2dem)
% 选择方差最大的非零输入作为耦合源
cands = {Pdist,'Pdist=Pws-dLe'; dLe,'dLe'; dLsteam,'dLsteam'; dH2dem,'dH2dem'};

for i=1:size(cands,1)
    v = cands{i,1};
    vars(i) = var(v);
end
[~,idx] = max(vars);
if vars(idx) < 1e-12
    % 全都几乎不变，退化处理：用一个很小噪声避免NaN（但提示无意义）
    xin = 1e-6*randn(size(Pdist));
    name = 'none(near-constant inputs)';
else
    xin = cands{idx,1};
    name = cands{idx,2};
end
end

function [lagSec, maxc] = lag_by_xcorr(x, y, fs, maxLag)
% 定义：lagSec > 0 表示 y 滞后 x（即 x 先变，y 后变）
% 用 xcorr(y,x) 的 lags 解释：若峰值出现在正 lag，表示 y 超前 x。
% 所以我们取 lagSec = -lag/fs 来满足“正值=输出滞后输入”。

[r,lags] = xcorr(y, x, maxLag, 'coeff');

% 若出现NaN（比如信号全0），做保护
if any(isnan(r))
    lagSec = NaN;
    maxc   = NaN;
    return;
end

[maxc, idx] = max(abs(r));   % 用绝对相关找最强耦合（符号可另存）
lag = lags(idx);
lagSec = -lag / fs;          % 关键：取负号使“正值=输出滞后输入”
end

function z = safe_ratio(a,b)
if ~isfinite(a) || ~isfinite(b) || abs(b)<1e-12
    z = NaN;
else
    z = a/b;
end
end

function BP = butter_bandpower(x, fs, f1, f2, order)
% BUTTER_BANDPOWER (robust)
% 频带能量/带通信号计算：
% - 默认用 butter + filtfilt 得到带通信号并算RMS
% - 若频段过低/过窄/数值病态风险高，则自动切换到频域积分法（无filtfilt）
%
% 输出：
% BP.band_rms, BP.band_power, BP.x_band, BP.method

if nargin < 5 || isempty(order), order = 4; end
x = x(:);

BP = struct('band_rms',NaN,'band_power',NaN,'x_band',zeros(size(x)),'method','');

if numel(x) < 64 || ~all(isfinite(x))
    BP.method = 'invalid_short';
    return;
end

nyq = fs/2;

% 保护频率范围
f1 = max(0, f1);
f2 = max(0, f2);
f1 = min(f1, nyq*0.999);
f2 = min(f2, nyq*0.999);

if f2 <= f1 || f2 <= 0
    BP.method = 'invalid_band';
    return;
end

% ========= 数值稳定性判据 =========
% 1) 归一化频率太接近0 或 带宽太窄 => filtfilt容易病态
W1 = f1/nyq;
W2 = f2/nyq;
relBW = (f2-f1)/max(f2,1e-9);

tooLow  = (W2 < 1e-3);          % 频段整体非常靠近0
tooNarrow = (relBW < 0.05);     % 相对带宽过窄

% 2) 信号几乎不变：滤波没意义，直接用频域
nearConst = (std(x) < 1e-8);

% 若满足任一风险条件，改用频域法
if tooLow || tooNarrow || nearConst
    [bp_rms, bp_pow] = bandpower_freqdomain(x, fs, f1, f2);
    BP.band_rms = bp_rms;
    BP.band_power = bp_pow;
    BP.x_band = zeros(size(x));     % 不提供时域带通信号
    BP.method = 'freqdomain';
    return;
end

% ========= 时域带通（butter + filtfilt） =========
try
    Wn = [W1 W2];
    [b,a] = butter(order, Wn, 'bandpass');
    % 去趋势再滤波，减小低频病态风险
    xf = filtfilt(b,a, detrend(x));
    BP.x_band = xf;
    BP.band_power = mean(xf.^2);
    BP.band_rms = sqrt(BP.band_power);
    BP.method = 'filtfilt';
catch
    % 若仍失败，降级为频域
    [bp_rms, bp_pow] = bandpower_freqdomain(x, fs, f1, f2);
    BP.band_rms = bp_rms;
    BP.band_power = bp_pow;
    BP.x_band = zeros(size(x));
    BP.method = 'freqdomain_fallback';
end
end

% ======== helper: 频域积分法（稳健，无filtfilt）========
function [rmsv, powv] = bandpower_freqdomain(x, fs, f1, f2)
x = detrend(x);

% 用 pwelch 比 periodogram 稳健
n = numel(x);
nfft = 2^nextpow2(min(max(256, n), 8192));
win = hamming(min(1024, n));
noverlap = round(0.5*length(win));

[Pxx,f] = pwelch(x, win, noverlap, nfft, fs);

idx = (f>=f1) & (f<=f2);
if ~any(idx)
    powv = 0;
else
    powv = trapz(f(idx), Pxx(idx));   % 功率（方差）
end
rmsv = sqrt(max(powv,0));
end
